# Task v1.5 — Terminal Mouse / Clipboard / IME / URL Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire mouse / clipboard / IME / URL paths between AppKit and libghostty in the macOS Caterm app so SSH-side users can select, copy, paste, scroll, ⌘-click URLs, drop files, and use IME — closing the gap left by Phase 1 (keyboard only).

**Architecture:** AppKit `NSView` overrides forward to a Swift wrapper around libghostty C APIs. libghostty owns selection, mouse-reporting, preedit, and URL detection. Clipboard is a bidirectional bridge: `GhosttyApp` runtime callbacks read/write `NSPasteboard.general` after a `DispatchQueue.main` hop; local paste vs. remote OSC 52 read is disambiguated by a `pendingLocalPaste` token raised on the surface before user-driven binding triggers.

**Tech Stack:** Swift 5.10, SwiftPM, AppKit, GhosttyKit (libghostty C bindings vendored as binary xcframework). Tests use XCTest in a new `TerminalEngineTests` target.

**Spec:** `docs/superpowers/specs/2026-04-29-task-v1.5-terminal-mouse-clipboard-ime-design.md` (commit `7c680d3`).

**Why these splits:**
- `InputMapping.swift` extracts the modifier / button / scroll mapping that was previously a private static helper on `GhosttySurface`. Pure functions with no AppKit dependencies → unit-testable.
- One `+<feature>.swift` extension per concern keeps each file under ~200 LOC and makes diff review straightforward.
- `pendingPasteText` / `pendingLocalPaste` are stored properties on the main `GhosttySurface` class (Swift extensions cannot add stored properties; the spec calls this out).
- File-drop handler lives in `GhosttySurfaceNSView+Drag.swift`, but it shares `pendingPasteText` plumbing with paste — that's why drag is in Task 2 (clipboard), not Task 5 (URL).

```swift
// apps/macos/Sources/TerminalEngine/InputMapping.swift — append

public func ghosttyMouseButton(buttonNumber: Int) -> ghostty_input_mouse_button_e {
    switch buttonNumber {
    case 0: return GHOSTTY_MOUSE_BUTTON_LEFT
    case 1: return GHOSTTY_MOUSE_BUTTON_RIGHT
    case 2: return GHOSTTY_MOUSE_BUTTON_MIDDLE
    default: return GHOSTTY_MOUSE_BUTTON_UNKNOWN
    }
}

public func scrollMods(precise: Bool, momentum: NSEvent.Phase) -> ghostty_input_scroll_mods_t {
    // libghostty packs these as bit flags. Bit 0 = precise; bits 1..3 reserved
    // for momentum phase per ghostty src/input/mouse.zig (see header line 122
    // comment). Keep the layout in sync if libghostty is upgraded.
    var bits: Int32 = 0
    if precise { bits |= 0x1 }
    switch momentum {
    case .began:      bits |= (1 << 1)
    case .changed:    bits |= (2 << 1)
    case .ended:      bits |= (3 << 1)
    case .cancelled:  bits |= (4 << 1)
    default: break
    }
    return ghostty_input_scroll_mods_t(bits)
}
```

> Note: the bit packing here is informed by the ghostty.h comment at line 122 ("This is a packed struct (see src/input/mouse.zig)..."). 5.3-OQ-1 covers verifying scroll behavior end-to-end via manual smoke; the unit test only validates the precise/imprecise bit.

### 1.6 — NSView mouse handlers

- [ ] **Step 1: Create `+Mouse` extension on the NSView**

```swift
// apps/macos/Sources/TerminalEngine/GhosttySurfaceNSView+Mouse.swift
import AppKit
import GhosttyKit

extension GhosttySurfaceNSView {

    public override func acceptsFirstMouse(for event: NSEvent?) -> Bool { true }

    public override func updateTrackingAreas() {
        super.updateTrackingAreas()
        for area in trackingAreas { removeTrackingArea(area) }
        let area = NSTrackingArea(
            rect: bounds,
            options: [
                .activeInKeyWindow,
                .mouseMoved,
                .mouseEnteredAndExited,
                .inVisibleRect,
                .cursorUpdate,
            ],
            owner: self,
            userInfo: nil
        )
        addTrackingArea(area)
    }

    public override func mouseDown(with e: NSEvent)        { forwardButton(e, .press, .left) }
    public override func mouseUp(with e: NSEvent)          { forwardButton(e, .release, .left) }
    public override func mouseDragged(with e: NSEvent)     { forwardPos(e) }
    public override func mouseMoved(with e: NSEvent)       { forwardPos(e) }
    public override func rightMouseDown(with e: NSEvent)   { forwardButton(e, .press, .right) }
    public override func rightMouseUp(with e: NSEvent)     { forwardButton(e, .release, .right) }
    public override func rightMouseDragged(with e: NSEvent){ forwardPos(e) }
    public override func otherMouseDown(with e: NSEvent)   { forwardButton(e, .press, ghosttyMouseButton(buttonNumber: e.buttonNumber)) }
    public override func otherMouseUp(with e: NSEvent)     { forwardButton(e, .release, ghosttyMouseButton(buttonNumber: e.buttonNumber)) }
    public override func otherMouseDragged(with e: NSEvent){ forwardPos(e) }

    public override func scrollWheel(with e: NSEvent) {
        guard let surface else { return }
        var dx = e.scrollingDeltaX
        var dy = e.scrollingDeltaY
        if !e.hasPreciseScrollingDeltas {
            dx *= surface.cellSize.width
            dy *= surface.cellSize.height
        }
        let mods = scrollMods(precise: e.hasPreciseScrollingDeltas, momentum: e.momentumPhase)
        surface.sendMouseScroll(deltaX: dx, deltaY: dy, mods: mods)
    }

    // MARK: - private

    private func forwardButton(_ e: NSEvent, _ state: ghostty_input_mouse_state_e, _ button: ghostty_input_mouse_button_e) {
        forwardPos(e)   // ensure libghostty has the latest position before the click
        _ = surface?.sendMouseButton(state: state, button: button, mods: e.modifierFlags)
    }

    private func forwardButton(_ e: NSEvent, _ state: ghostty_input_mouse_state_e, _ named: NamedButton) {
        let button: ghostty_input_mouse_button_e = (named == .left)
            ? GHOSTTY_MOUSE_BUTTON_LEFT
            : GHOSTTY_MOUSE_BUTTON_RIGHT
        forwardButton(e, state, button)
    }

    private func forwardPos(_ e: NSEvent) {
        guard let surface else { return }
        let p = convert(e.locationInWindow, from: nil)
        surface.sendMousePos(x: Double(p.x), y: Double(p.y), mods: e.modifierFlags)
    }

    private enum NamedButton { case left, right }
}
```

```swift
    _, kind, contentsPtr, count, _ in
    guard kind == GHOSTTY_CLIPBOARD_STANDARD,
          let contentsPtr else { return }

    var picked: String?
    for i in 0..<Int(count) {
        let entry = contentsPtr[i]

        let mime = entry.mime.map { String(cString: $0) } ?? ""
```

```swift
public static func writeManagedConfig() throws {
    let path = managedConfigPath
    try FileManager.default.createDirectory(
```

```swift
// apps/macos/Sources/TerminalEngine/GhosttySurfaceNSView+URL.swift
import AppKit
import GhosttyKit

extension GhosttySurfaceNSView {

    func wireURLHandlers() {
        guard let surface else { return }
        surface.onHoverURL = { [weak self] url in
            self?.hoveredURL = url
            self?.window?.invalidateCursorRects(for: self!)
        }
        surface.onOpenURL = { [weak self] urlString, _ in
            self?.openURL(urlString)
        }
    }

    fileprivate func openURL(_ urlString: String) {
        guard let url = URL(string: urlString),
              let scheme = url.scheme
        else { return }

        if isSafeURLScheme(scheme) {
            NSWorkspace.shared.open(url)
            return
        }

        guard let win = window else { return }
        let alert = NSAlert()
        alert.messageText = "Open this URL?"
        alert.informativeText = "The URL uses scheme “\(scheme)” which Caterm doesn't open by default:\n\n\(urlString)"
        alert.alertStyle = .warning
        alert.addButton(withTitle: "Cancel")     // default
        alert.addButton(withTitle: "Open")
        alert.beginSheetModal(for: win) { resp in
            if resp == .alertSecondButtonReturn {
                NSWorkspace.shared.open(url)
            }
        }
    }
}
```
